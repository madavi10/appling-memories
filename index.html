  <script>

    // -------------- CONFIG --------------

    const CATEGORIES = [

      "All Videos",

      "Family",

      "Andrew",

      "Matthew",

      "Michael",

      "Mary Kay"

    ];


    const XML_PATH = "videos.xml"; // must be in same folder or adjust this path


    // -------------- STATE --------------

    let keywordMap = {};   // { categoryName: [keywords...] }

    let videos = [];       // [{title, filename, url, description}]

    let activeCategory = "All Videos";

    let searchTerm = "";


    // -------------- INIT --------------

    document.addEventListener("DOMContentLoaded", () => {

      buildBladeTabs();

      attachSearchListener();

      loadXmlConfig();

    });


    // -------------- UI BUILDERS --------------

    function buildBladeTabs() {

      const container = document.getElementById("bladeTabs");

      container.innerHTML = "";


      CATEGORIES.forEach((name, index) => {

        const tab = document.createElement("div");

        tab.className = "blade-tab" + (index === 0 ? " active" : "");

        tab.dataset.category = name;


        const label = document.createElement("span");

        label.textContent = name.toLowerCase(); // match Xbox style

        tab.appendChild(label);


        tab.addEventListener("click", () => {

          setActiveCategory(name);

        });


        container.appendChild(tab);

      });

    }


    function attachSearchListener() {

      const input = document.getElementById("searchInput");

      input.addEventListener("input", (e) => {

        searchTerm = e.target.value.trim().toLowerCase();

        renderVideoList();

      });

    }


    function setActiveCategory(categoryName) {

      activeCategory = categoryName;


      // Update blade UI

      document.querySelectorAll(".blade-tab").forEach(tab => {

        tab.classList.toggle("active", tab.dataset.category === categoryName);

      });


      // Update header meta

      document.getElementById("currentCategoryLabel").textContent = categoryName;

      renderVideoList();

    }


    // -------------- XML LOADING & PARSING --------------

    async function loadXmlConfig() {

      try {

        const response = await fetch(XML_PATH);

        if (!response.ok) {

          throw new Error("Could not load videos.xml");

        }


        const xmlText = await response.text();

        const parser = new DOMParser();

        const xmlDoc = parser.parseFromString(xmlText, "application/xml");


        parseKeywords(xmlDoc);

        parseVideos(xmlDoc);


        renderVideoList();

      } catch (err) {

        console.error(err);

        document.getElementById("videoCount").textContent =

          "Error loading videos.xml – check path and hosting.";

      }

    }


    function parseKeywords(xmlDoc) {

      keywordMap = {};

      const categoryNodes = xmlDoc.querySelectorAll("keywords > category");


      categoryNodes.forEach(catNode => {

        const name = catNode.getAttribute("name");

        const keywords = Array.from(catNode.getElementsByTagName("keyword"))

          .map(k => k.textContent.trim())

          .filter(Boolean);


        keywordMap[name] = keywords;

      });


      // All Videos doesn't need keywords

      if (!keywordMap["All Videos"]) {

        keywordMap["All Videos"] = [];

      }

    }


    function parseVideos(xmlDoc) {

      videos = [];

      const videoNodes = xmlDoc.querySelectorAll("videos > video");


      videoNodes.forEach(node => {

        const title = getText(node, "title");

        const filename = getText(node, "filename");

        const url = getText(node, "url");

        const description = getText(node, "description");


        if (url) {

          videos.push({

            title: title || filename || "Untitled Video",

            filename: filename || "",

            url,

            description: description || ""

          });

        }

      });


      function getText(parent, tagName) {

        const el = parent.getElementsByTagName(tagName)[0];

        return el ? el.textContent.trim() : "";

      }

    }


    // -------------- FILTERING --------------

    function filterVideos() {

      let filtered = videos.slice();


      if (activeCategory !== "All Videos") {

        const keywords = (keywordMap[activeCategory] || []).map(k => k.toLowerCase());

        if (keywords.length > 0) {

          filtered = filtered.filter(video => {

            const text = (video.filename + " " + video.title).toLowerCase();

            return keywords.some(kw => text.includes(kw));

          });

        } else {

          // If category has no keywords configured, show nothing (or everything).

          filtered = [];

        }

      }


      if (searchTerm) {

        filtered = filtered.filter(video => {

          const text = (video.title + " " + video.filename + " " + video.description).toLowerCase();

          return text.includes(searchTerm);

        });

      }


      return filtered;

    }


    // -------------- RENDER --------------

    function renderVideoList() {

      const listEl = document.getElementById("videoList");

      const countEl = document.getElementById("videoCount");


      const filtered = filterVideos();

      listEl.innerHTML = "";


      if (filtered.length === 0) {

        countEl.textContent = "No videos found for this view.";

        return;

      }


      countEl.textContent = `${filtered.length} video${filtered.length !== 1 ? "s" : ""} found`;


      filtered.forEach(video => {

        const item = document.createElement("div");

        item.className = "video-item";

        item.addEventListener("click", () => openVideo(video.url));


        const main = document.createElement("div");

        main.className = "video-main";


        const title = document.createElement("div");

        title.className = "video-title";

        title.textContent = video.title;

        main.appendChild(title);


        if (video.filename) {

          const fn = document.createElement("div");

          fn.className = "video-filename";

          fn.textContent = video.filename;

          main.appendChild(fn);

        }


        if (video.description) {

          const desc = document.createElement("div");

          desc.className = "video-description";

          desc.textContent = video.description;

          main.appendChild(desc);

        }


        const actions = document.createElement("div");

        actions.className = "video-actions";


        const btn = document.createElement("button");

        btn.className = "play-button";

        btn.textContent = "Play";

        btn.addEventListener("click", (e) => {

          e.stopPropagation(); // avoid double-trigger

          openVideo(video.url);

        });


        const tags = document.createElement("div");

        tags.className = "video-tags";

        tags.textContent = "Opens in Google Drive";


        actions.appendChild(btn);

        actions.appendChild(tags);


        item.appendChild(main);

        item.appendChild(actions);


        listEl.appendChild(item);

      });

    }


    function openVideo(url) {

      window.open(url, "_blank", "noopener");

    }
